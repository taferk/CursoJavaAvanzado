2. PersonaServicio.java ‚Äì Servicio REST

// PersonaServicio.java
package com.ejemplo.servicio;

import com.ejemplo.modelo.Persona;
import jakarta.ws.rs.GET;
import jakarta.ws.rs.Path;
import jakarta.ws.rs.Produces;
import jakarta.ws.rs.core.MediaType;
import java.util.List;

@Path("/personas")
public class PersonaServicio {

    @GET
    @Produces(MediaType.APPLICATION_JSON)
    public List<Persona> obtenerPersonas() {
        return List.of(
            new Persona("Ana", 30),
            new Persona("Luis", 25)
        );
    }
}

ü§ù Cliente Java

Usamos HttpClient para consumir el servicio REST.

// ClienteRest.java
import java.net.URI;
import java.net.http.*;
import java.io.IOException;

public class ClienteRest {
    public static void main(String[] args) throws IOException, InterruptedException {
        HttpClient cliente = HttpClient.newHttpClient();
        HttpRequest solicitud = HttpRequest.newBuilder()
            .uri(URI.create("http://localhost:8080/tuApp/api/personas"))
            .GET()
            .build();

        HttpResponse<String> respuesta = cliente.send(solicitud, HttpResponse.BodyHandlers.ofString());
        System.out.println("Respuesta JSON:");
        System.out.println(respuesta.body());
    }
}
